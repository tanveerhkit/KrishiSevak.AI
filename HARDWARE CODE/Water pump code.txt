
#include <esp_now.h>
#include <WiFi.h>
#include <HardwareSerial.h>

// Define pins and GSM module serial connection
const int ledPin = 2; // Onboard LED pin
const int relayPin = 13; // GPIO pin where the relay is connected

HardwareSerial gsmSerial(1); // Initialize HardwareSerial for GSM
const int RX_PIN = 16; // ESP32 RX pin to GSM module TX
const int TX_PIN = 17; // ESP32 TX pin to GSM module RX
const char phoneNumber[] = "+919653030160"; // Replace with your phone number

typedef struct struct_message {
    bool state; // true = soil wet, false = soil dry
} struct_message;

struct_message stateIn;
esp_now_peer_info_t peerInfo;

bool lastRelayState = LOW; // Keep track of the last relay state (reversed logic)

// Function to send SMS
void sendSMS(const char* message) {
  gsmSerial.println("AT+CMGF=1"); // Set SMS mode to text
  delay(100);
  gsmSerial.print("AT+CMGS=\"");
  gsmSerial.print(phoneNumber);
  gsmSerial.println("\"");
  delay(100);
  gsmSerial.print(message);
  gsmSerial.write(26); // ASCII code for CTRL+Z to send the SMS
  delay(1000);
}

// Callback when data is received
void OnDataRecv(const esp_now_recv_info *info, const uint8_t *incomingData, int len) {
  memcpy(&stateIn, incomingData, sizeof(stateIn));
  Serial.print("Data Packet received from krishi seva Module 1: ");
  Serial.println(len);

  // Reverse logic: Control LED and relay based on received data
  if (!stateIn.state) {
    // Soil is dry
    digitalWrite(ledPin, HIGH); // Turn on the LED
    digitalWrite(relayPin, LOW); // Turn ON the relay (motor OFF) - reversed logic
    Serial.println("Your Plants are happy <3."); // Swapped message

    // Send SMS if relay state has changed
    if (lastRelayState != HIGH) {
      sendSMS("Motor has been turned ON (reversed logic).");
      lastRelayState = HIGH;
    }
  } else {
    // Soil is wet
    digitalWrite(ledPin, LOW); // Turn off the LED
    digitalWrite(relayPin, HIGH); // Turn OFF the relay (motor ON) - reversed logic
    Serial.println("Oops! your Plant needs water, Irrigation is Start."); // Swapped message

    // Send SMS if relay state has changed
    if (lastRelayState != LOW) {
      sendSMS("Motor has been turned OFF (reversed logic).");
      lastRelayState = LOW;
    }
  }
}

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(relayPin, OUTPUT);
  digitalWrite(ledPin, LOW); // Initialize LED to be off
  digitalWrite(relayPin, HIGH); // Initialize relay to be OFF (motor ON - reversed logic)

  // Initialize Serial Monitor
  Serial.begin(115200);

  // Initialize GSM Serial Communication
  gsmSerial.begin(9600, SERIAL_8N1, RX_PIN, TX_PIN);

  // Set device as a Wi-Fi Station
  WiFi.mode(WIFI_STA);

  // Init ESP-NOW
  if (esp_now_init() != ESP_OK) {
    Serial.println("Error initializing ESP-NOW");
    return;
  }

  // Register for a callback function that will be called when data is received
  esp_now_register_recv_cb(OnDataRecv);

  // Register peer
  uint8_t broadcastAddress[] = {0x24, 0x62, 0xAB, 0xE0, 0xE7, 0x94}; // Replace with Sensor ESP32 MAC Address
  memcpy(peerInfo.peer_addr, broadcastAddress, 6);
  peerInfo.channel = 0;
  peerInfo.encrypt = false;

  // Add peer
  if (esp_now_add_peer(&peerInfo) != ESP_OK) {
    Serial.println("Failed to add peer");
    return;
  }
}

void loop() {
  // No code needed here, LED, relay, and SMS sending are handled in the callback function
}
